### 💡 กลยุทธ์การ Optimize & Save Tokens (เพื่อแก้ปัญหา Token Error)

ปัญหา "Token ใช้งานไม่ได้" มักเกิดจาก 2 สาเหตุ: **Context ล้น** หรือ **Input ยาวเกินความจำเป็น** วิธีแก้ที่ผมจะใส่ไว้ใน Prompt คือ:

1. **Summary-as-History:** แทนที่จะส่งประวัติการคุยทั้งหมดกลับไป Agent จะต้องมีหน้าที่ "สรุปความคืบหน้า" เป็นระยะเพื่อทิ้งรายละเอียดที่ไม่จำเป็น
2. **Selective Context:** อ่านเฉพาะส่วนที่สำคัญของไฟล์ (เช่น หัวไฟล์/ท้ายไฟล์) แทนการอ่านทั้งไฟล์ถ้าไม่จำเป็น
3. **Strict JSON Output:** บังคับให้ AI ตอบกลับด้วยรูปแบบที่สั้นและกระชับที่สุด ลด "คำพ่น" (Verbose text)

3. โค้ดส่วน Backend (server.js - The Gatekeeper)
เราต้องสร้างฟังก์ชันคอยดักจับก่อนเรียกใช้เครื่องมือจริง:

JavaScript
// backend/src/server.js

async function executeWithSecurity(toolName, args, isAutoAllow) {
    const dangerousTools = ['organize_file', 'delete_file', 'write_file'];
    
    // บันทึก Log เริ่มต้น
    const logId = addActivityLog(toolName, args, 'Pending');

    if (dangerousTools.includes(toolName) && !isAutoAllow) {
        // ส่งสัญญาณไปที่ Frontend ผ่าน SSE เพื่อรอการกดยืนยัน
        sendSSE('REQUIRE_APPROVAL', { logId, toolName, args });
        
        // รอการตอบกลับจาก User (ใช้ Promise หรือ Long Polling)
        const userDecision = await waitForUserDecision(logId);
        
        if (userDecision === 'REJECTED') {
            updateActivityLog(logId, 'Blocked', 'Human');
            throw new Error(`Action ${toolName} was blocked by user.`);
        }
        updateActivityLog(logId, 'Approved', 'Human');
    } else {
        updateActivityLog(logId, 'Auto-Approved', 'System');
    }

    // รันเครื่องมือจริง
    return await tools[toolName](args);
}

1. โค้ด Backend: ฟังก์ชัน read_activity_log
เพิ่มฟังก์ชันนี้เข้าไปใน backend/src/tools/fileManager.js หรือไฟล์ที่คุณเก็บ Tool ไว้ครับ:

JavaScript
const fs = require('fs');
const path = require('path');

// ฟังก์ชันสำหรับอ่าน Log ล่าสุดเพื่อให้ AI วิเคราะห์
async function read_activity_log(limit = 10) {
    try {
        const logPath = path.join(__dirname, '../../workspace/activity.log');
        
        if (!fs.existsSync(logPath)) {
            return "No activity logs found yet. The system is fresh.";
        }

        const data = fs.readFileSync(logPath, 'utf8');
        const lines = data.trim().split('\n');
        
        // อ่านเฉพาะบรรทัดท้ายๆ ตามจำนวน limit เพื่อประหยัด Token
        const recentLogs = lines.slice(-limit).map(line => JSON.parse(line));
        
        return JSON.stringify({
            status: "success",
            message: `Retrieved last ${recentLogs.length} activities`,
            data: recentLogs
        });
    } catch (error) {
        return JSON.stringify({ status: "error", message: error.message });
    }
}
